// Get current time in seconds
const getCurrentTimeInSeconds = () => {
    const now = new Date();
    return now.getHours() * 3600 + now.getMinutes() * 60 + now.getSeconds();
};

// Get current day
const getCurrentDay = () => {
    const days = [
        "SUNDAY",
        "MONDAY",
        "TUESDAY",
        "WEDNESDAY",
        "THURSDAY",
        "FRIDAY",
        "SATURDAY",
    ];
    return days[new Date().getDay()];
};

// Check if restaurant is open now
const isRestaurantOpen = (operatingHours) => {
    const currentDay = getCurrentDay();
    const todayHours = operatingHours.find((hour) => hour.day === currentDay);

    if (!todayHours || !todayHours.isOpen) {
        return false;
    }

    if (!todayHours.openTime || !todayHours.closeTime) {
        return true; // If no time specified, assume open all day
    }

    const currentTimeSeconds = getCurrentTimeInSeconds();
    const [openHour, openMin] = todayHours.openTime.split(":").map(Number);
    const [closeHour, closeMin] = todayHours.closeTime.split(":").map(Number);

    const openTimeSeconds = openHour * 3600 + openMin * 60;
    const closeTimeSeconds = closeHour * 3600 + closeMin * 60;

    if (openTimeSeconds <= closeTimeSeconds) {
        // Normal case (9AM → 5PM)
        return (
            currentTimeSeconds >= openTimeSeconds &&
            currentTimeSeconds <= closeTimeSeconds
        );
    } else {
        // Cross midnight case (10PM → 3AM)
        return (
            currentTimeSeconds >= openTimeSeconds ||
            currentTimeSeconds <= closeTimeSeconds
        );
    }
};

const setupStore = async (userId, data, files) => {
  // Check if user exists
  const user = await authPartnerModel.findUserById(userId);
  if (!user) {
    throw new AppError("User not found", 404, ERROR_CODES.USER_NOT_FOUND);
  }

  if (user.userType !== "buissness") {
    throw new AppError(
      "Only business users can create store",
      403,
      ERROR_CODES.FORBIDDEN
    );
  }

  // Extract fields
  const {
    name,
    email,
    phone,
    description,
    taxId,
    buissnessLicenceId,
    storeLocation,
    country,
    city,
    state,
    address,
    latitude,
    longitude,
    instructions,
    deliveryType,
    deliveryTime,
    categoryIds,
    wholeWeekSameTime,
    operatingHours,
    openTime,
    closeTime,
  } = data;

  // Validate required fields
  if (!name || !email || !phone) {
    throw new AppError(
      "Name, email, and phone are required",
      400,
      ERROR_CODES.INVALID_INPUT
    );
  }

  // Check store email uniqueness (globally)
  if (email) {
    const emailExists = await prisma.store.findFirst({ where: { email, isDeleted: false } });
    if (emailExists) {
      throw new AppError(
        "Store email already exists",
        409,
        ERROR_CODES.STORE_EMAIL_EXISTS
      );
    }
  }

  // Check store phone uniqueness (globally)
  if (phone) {
    const phoneExists = await prisma.store.findFirst({ where: { phone, isDeleted: false } });
    if (phoneExists) {
      throw new AppError(
        "Store phone already exists",
        409,
        ERROR_CODES.STORE_PHONE_EXISTS
      );
    }
  }

  // Check tax ID uniqueness (globally)
  if (taxId) {
    const taxExists = await prisma.store.findFirst({ where: { taxId, isDeleted: false } });
    if (taxExists) {
      throw new AppError(
        "Tax ID already exists",
        409,
        ERROR_CODES.TAX_ID_EXISTS
      );
    }
  }

  // Check business license ID uniqueness (globally)
  if (buissnessLicenceId) {
    const licenseExists = await authPartnerModel.findStoreByBusinessLicenseId(
      buissnessLicenceId
    );
    if (licenseExists) {
      throw new AppError(
        "Business License ID already exists",
        409,
        ERROR_CODES.BUSINESS_LICENSE_EXISTS
      );
    }
  }

  // Parse category IDs
  let parsedCategoryIds = [];
  if (categoryIds) {
    if (Array.isArray(categoryIds)) {
      parsedCategoryIds = categoryIds;
    } else if (typeof categoryIds === "string") {
      if (categoryIds.includes(",")) {
        parsedCategoryIds = categoryIds.split(",").map((id) => id.trim());
      } else {
        try {
          const parsed = JSON.parse(categoryIds);
          // 🔥 FIX
          parsedCategoryIds = Array.isArray(parsed) ? parsed : [parsed];
        } catch (err) {
          // If not JSON, treat as single ID string
          parsedCategoryIds = [categoryIds];
        }
      }
    }

    // Validate category IDs exist in database
    if (parsedCategoryIds.length > 0) {
      const foundCategories = await authPartnerModel.getStoreCategoriesById(parsedCategoryIds);

      if (foundCategories.length !== parsedCategoryIds.length) {
        const foundIds = foundCategories.map((c) => c.id);
        const invalidIds = parsedCategoryIds
          .map((id) => parseInt(id))
          .filter((id) => !foundIds.includes(id));
        throw new AppError(
          `Invalid category IDs: ${invalidIds.join(", ")}. These categories do not exist.`,
          400,
          ERROR_CODES.INVALID_INPUT
        );
      }
    }
  }

  // Parse operating hours (same logic as restaurant)
  let parsedOperatingHours = [];
  const useSameTimingBool = wholeWeekSameTime === "true" || wholeWeekSameTime === true;

  if (useSameTimingBool) {
    if (!openTime || !closeTime) {
      throw new AppError(
        "openTime and closeTime are required when wholeWeekSameTime is true",
        400,
        "INVALID_INPUT"
      );
    }

    const allDays = [
      "MONDAY",
      "TUESDAY",
      "WEDNESDAY",
      "THURSDAY",
      "FRIDAY",
      "SATURDAY",
      "SUNDAY",
    ];

      day,
      isOpen: true,
      openTime: openTime.trim(),
      closeTime: closeTime.trim(),
    }));
  } else {
    if (typeof operatingHours === "string") {
      try {
        parsedOperatingHours = JSON.parse(operatingHours);

        throw new AppError(

  }

  // Handle video (single optional file)
  if (files?.video?.[0]) {
    storeData.video = `${baseURL}/uploads/videos/${files.video[0].filename}`;
  }

  // Create store
  const store = await authPartnerModel.createStore(storeData);

  // Create operating hours
  if (parsedOperatingHours && parsedOperatingHours.length > 0) {
    await authPartnerModel.upsertStoreOperatingHours(
      store.id,
      parsedOperatingHours
    );
  }

  // Create store categories
  if (parsedCategoryIds.length) {
    const categoryIdNumbers = parsedCategoryIds.map((id) => Number(id));
    await authPartnerModel.upsertStoreCategories(store.id, categoryIdNumbers);
  }

  // Fetch complete store data
  const completeStore = await authPartnerModel.findStoreById(store.id);
  const updatedUser = await authPartnerModel.findUserById(userId);

  // Clean user data
  const { passwordHash, ...safeUser } = updatedUser;

  const { storeCategories, ...storeWithoutCategories } = completeStore;
  const transformedStore = {
    ...storeWithoutCategories,
    images: storeWithoutCategories.images
      ? storeWithoutCategories.images.split(",")
      : [],
    buissnessLicenseImg: storeWithoutCategories.buissnessLicenseImg
      ? [storeWithoutCategories.buissnessLicenseImg]
      : [],
    categories: storeCategories?.map((sc) => ({
      id: sc.category.id,
      name: sc.category.name,
    })) || [],
  };
  let parsedOwnerLicenseImg = null;

  if (safeUser.OwnerLicenseImg) {
    try {
      parsedOwnerLicenseImg = JSON.parse(safeUser.OwnerLicenseImg);
    } catch (err) {
      parsedOwnerLicenseImg = null;
    }
  }

  const responseData = {
    ...safeUser,
    OwnerLicenseImg: parsedOwnerLicenseImg,
    store: transformedStore,
  };

  // Remove null/undefined
  Object.keys(responseData).forEach(
    (key) => responseData[key] == null && delete responseData[key]
  );

  return responseData;
};
